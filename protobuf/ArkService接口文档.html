ArkService 是 ArkSim 仿真引擎的外部控制接口，支持 JSON 和 Protobuf 两种协议格式。

一、仿真控制接口 JSON

请求格式：{"fn": "<命令>", ...}，需要 uuid 的场景由 arkservice 回传。

fn参数说明
start args.exec: int
args.offscreen: bool
args.randomSeed: int
args.realtime: bool
args.scenarios: string[]
args.simType: int 		启动一个仿真实例
resumeuuid: string继续仿真（uuid 换成 arkservice 回传的 uuid）
pauseuuid: string暂停仿真
stopuuid: string停止仿真
restartuuid: string重推仿真
runstepargs.step: int, uuid: string步进仿真
advance_to_timeargs.time: double, uuid: string快进到某一时刻
set_clock_rateargs.rate: double, uuid: string设置仿真倍速
proto proto: string, uuid: string 实体控制指令。字符串由 ArkSimActionProto.ActionsFromOutside 消息赋值后进行序列化（详见 第三节）。态势类数据详见 zmq_observer.ag

start 参数说明

参数类型说明
execint执行模式（1=执行）
offscreenbool是否离屏渲染
randomSeedint随机种子（0=默认）
realtimebool是否实时模式
scenariosstring[]场景文件路径列表
simTypeint仿真类型

示例

// 启动仿真
{"args":{"exec":1,"offscreen":false,"randomSeed":0,"realtime":false,
 "scenarios":["D:/ArkSIM3.1.6/demo/floridistan/floridistan.txt"],"simType":0},"fn":"start"}

// 继续
{"fn":"resume","uuid":"d7ef233bb7d3489c911dc95d89f8df22"}

// 暂停
{"fn":"pause","uuid":"d7ef233bb7d3489c911dc95d89f8df22"}

// 停止
{"fn":"stop","uuid":"d7ef233bb7d3489c911dc95d89f8df22"}

// 重推
{"fn":"restart","uuid":"d7ef233bb7d3489c911dc95d89f8df22"}

// 步进
{"fn":"runstep","args":{"step":1},"uuid":"d7ef233bb7d3489c911dc95d89f8df22"}

// 快进到某一时刻
{"fn":"advance_to_time","args":{"time":1.0},"uuid":"d7ef233bb7d3489c911dc95d89f8df22"}

// 倍速
{"fn":"set_clock_rate","args":{"rate":1.0},"uuid":"d7ef233bb7d3489c911dc95d89f8df22"}

// 实体控制
{"fn":"proto","proto":"<ActionsFromOutside序列化字符串>","uuid":"d7ef233bb7d3489c911dc95d89f8df22"}

二、态势切换与输出控制接口 JSON

以下接口于 2024 年 11 月 1 日添加。请求格式：{"fn": "<命令>", ..., "uuid": "<uuid>"}

fn参数说明
changesituation rate: int (0 / 1)
uuid: string		 切换态势类型：
• 0 — 定制态势，对应 arksim_customized_situation.proto 中结构
• 1 — 实时变化战场态势，对应 zmq_observer_pb3.proto 中结构
simulationtimeswitch rate: bool
uuid: string		 开关仿真时间输出（默认开启）
customizedsituation time: double
uuid: string		 设置定制态势输出时间间隔（默认 3 秒）

示例

// 切换态势类型 — 定制态势
{"fn":"changesituation","rate":0,"uuid":"d7ef233bb7d3489c911dc95d89f8df22"}

// 开关仿真时间输出 — 关闭
{"fn":"simulationtimeswitch","rate":false,"uuid":"d7ef233bb7d3489c911dc95d89f8df22"}

// 设置定制态势输出时间间隔 — 3秒
{"fn":"customizedsituation","time":3.0,"uuid":"d7ef233bb7d3489c911dc95d89f8df22"}

三、实体控制接口 Protobuf

定义文件：arksimActions.proto，package ArkSimActionProto。实体控制指令打包在 ActionsFromOutside 消息中，序列化后通过 {"fn":"proto", "proto":"<序列化字符串>", ...} 发送。

3.1 控制指令枚举 E_Actions

枚举值编号说明
E_SetAgentOutsideControl0设置可控制实体
E_ReleaseOutsideControl1取消可控制实体
E_SetDesiredVelocity2设置期望速度
E_SetDesiredAltitude3设置期望高度
E_SetDesiredHeading4设置期望朝向
E_GoToLocation5改变位置
E_FollowRoute6设置跟随路线
E_AuxActions7AUX 指令（透传至脚本处理）
E_TurnOnSensor8打开传感器
E_TurnOffSensor9关闭传感器
E_ChangeSensorMode10改变传感器工作模式
E_GetSensorCurrentMode11获取传感器工作模式
E_FireAtTarget12开火
E_FireSlavoAtTarget13用特定武器向目标齐射
E_UpdateTarget14更新当前目标
E_StartJamming15开干扰
E_StopJamming16关干扰
E_ChangeJammingMode17改变干扰工作模式
E_TurnOnComm18通信开机
E_TurnOffComm19通信关机
E_SendMsgToPlatform20向特定平台发送消息
E_SendMsgToCommandChain21向特定指挥链发送消息
E_ChangePlatformNumber22修改仿真实体数量（增加实体和删除实体）
E_ChangeCommander23更换实体的指挥链上级

3.2 顶层消息 ActionsFromOutside

message ActionsFromOutside {
    repeated AgentContrl            a_agentcontrl        = 1;
    repeated DesiredHeading         a_desiredheading      = 2;
    repeated DesiredAltitude        a_desiredaltitude     = 3;
    repeated DesiredVelocity        a_desiredvelocity     = 4;
    repeated GoToLocation           a_gotolocation        = 5;
    repeated FollowRoute            a_followroute         = 6;
    repeated SensorAction           a_sensoraction        = 7;
    repeated ChangeSensorMode       a_changesensormode    = 8;
    repeated FireAtTarget           a_fireattarget        = 9;
    repeated FireSlavoAtTarget      a_fireslavoattarget   = 10;
    repeated ChangeJammingMode      a_changejammingmode   = 11;
    repeated SendMsgToPlatform      a_sendmsgtoplatform   = 12;
    repeated SendMsgToCommandChain  a_sendmsgtocommandchain = 13;
    repeated AfsimAuxCommand        a_afsimauxcommand     = 14;
    repeated ChangeCommander        a_ChangeCommander     = 15;
    repeated ChangePlatformNumber   a_ChangePlatformNumber = 16;
}

3.3 基础消息类型

AgentName

message AgentName {
    string agent_id     = 1; // 目标实体的名字
    string Component_id = 2; // 目标携带传感器、通信组件、武器的名字（仅控制实体时为空）
}

3.4 运动控制

DesiredHeading（指令 4）

message DesiredHeading {
    string agent_id                = 1; // 目标实体名字
    double desired_heading         = 2; // 要求 platform 转向的 heading（radians），顺时针为正，北=0
    bool   has_desired_velocity    = 3; // 转弯时使用的速度 是否指定（可选）
    double desired_velocity        = 4; // 期望转弯速度（m/s）
    bool   has_desired_turn_direction = 5; // 转向方向 是否指定（可选，不指定则仿真采用最短路径）
    uint32 desired_turn_direction  = 6; // 0=左，1=右
}

DesiredAltitude（指令 3）

message DesiredAltitude {
    string agent_id                = 1; // 目标实体名字
    double desired_altitude        = 2; // 期望高度（m）
    bool   has_desired_altitude_rate = 3; // 上升/下降速率 是否指定（可选）
    double desired_altitude_rate   = 4; // 期望高度变化速率（m/s，如果指定总是正值）
}

DesiredVelocity（指令 2）

message DesiredVelocity {
    string agent_id         = 1; // 目标实体名字
    double desired_velocity = 2; // 期望速度
    double linearAccel      = 3; // 线性加速度
}

GoToLocation（指令 5）

message GoToLocation {
    string          agent_id            = 1; // 目标实体名字
    uint32          priority            = 2; // 优先级
    repeated double reportedLocationLLA = 3; // 目标位置 LLA [纬度, 经度, 高度]
}

FollowRoute（指令 6）

message Waypoint {
    string          WaypointID          = 1;
    double          speed               = 2;
    repeated double reportedLocationLLA = 3; // [纬度, 经度, 高度]
}

message FollowRoute {
    string             agent_id       = 1; // 目标实体名字
    string             aRouteName     = 2; // 路线名称
    repeated Waypoint  repeatedpoint  = 3; // 路径点集合
}

AgentContrl（指令 0, 1）

message AgentContrl {
    E_Actions action   = 1; // E_SetAgentOutsideControl / E_ReleaseOutsideControl
    string    agent_id = 2; // 目标实体名字
}

四、态势输出数据结构 Protobuf

定义文件：arksimproto.proto，package ArkSIMCustomizedSituation。切换态势类型使用 changesituation 接口（见 第二节）：0 对应本文件结构，1 对应 zmq_observer_pb3.proto。
